**Important \#1: Sponsor Relay Hardening**

**Files:** api-server/index.ts

We are implementing in-memory rate limiting to prevent "Presence Hammering" and ensuring that signals have a 10-minute TTL (Time-To-Live) so users don't receive stale support signals.

TypeScript

// api-server/index.ts  
import express from 'express';  
const app \= express();

// In-memory store for rate limiting: { \[token/IP\]: { count: number, reset: number } }  
const rateLimits: Record\<string, { count: number, reset: number }\> \= {};  
const globalIPLimits: Record\<string, { count: number, reset: number }\> \= {};

const checkRateLimit \= (key: string, limit: number, windowMs: number) \=\> {  
  const now \= Date.now();  
  if (\!rateLimits\[key\] || now \> rateLimits\[key\].reset) {  
    rateLimits\[key\] \= { count: 1, reset: now \+ windowMs };  
    return { allowed: true };  
  }  
  rateLimits\[key\].count++;  
  return {   
    allowed: rateLimits\[key\].count \<= limit,   
    remaining: Math.max(0, limit \- rateLimits\[key\].count),  
    reset: rateLimits\[key\].reset   
  };  
};

// Middleware: Global IP Cap (200 req/min)  
app.use((req, res, next) \=\> {  
  const ip \= req.ip;  
  const now \= Date.now();  
  if (\!globalIPLimits\[ip\] || now \> globalIPLimits\[ip\].reset) {  
    globalIPLimits\[ip\] \= { count: 1, reset: now \+ 60000 };  
  } else {  
    globalIPLimits\[ip\].count++;  
    if (globalIPLimits\[ip\].count \> 200) return res.status(429).send('Too Many Requests');  
  }  
  next();  
});

app.get('/sponsor/poll', (req, res) \=\> {  
  const token \= req.headers\['x-sponsor-token'\] as string;  
  const limit \= checkRateLimit(\`poll:${token}\`, 20, 60000);

  if (\!limit.allowed) {  
    res.set('Retry-After', String(Math.ceil((limit.reset \- Date.now()) / 1000)));  
    return res.status(429).json({ error: 'Rate limit exceeded: 20 polls/min' });  
  }

  // Signal Expiry Logic: Purge signals older than 10 mins  
  const tenMinsAgo \= Date.now() \- 600000;  
  channel.queue \= channel.queue.filter(sig \=\> sig.ts \> tenMinsAgo);

  res.json(channel.queue);  
});

app.post('/sponsor/check-in', (req, res) \=\> {  
  const token \= req.headers\['x-sponsor-token'\] as string;  
  const limit \= checkRateLimit(\`checkin:${token}\`, 5, 60000);

  if (\!limit.allowed) return res.status(429).json({ error: 'Rate limit exceeded: 5 check-ins/min' });

  // Add timestamp to new signals  
  const signal \= { ...req.body, ts: Date.now() };  
  channel.queue.push(signal);  
  res.sendStatus(200);  
});

**CTO Confirmation:** When a client hits the limit, they receive an **HTTP 429** with a Retry-After header. The poll response only contains "fresh" signals from the last 10 minutes.

// components/BurningRitual.tsx  
export const BurningRitual \= () \=\> {  
  const \[isConfirming, setIsConfirming\] \= useState(false);

  const onHoldComplete \= () \=\> setIsConfirming(true);

      \<View style\={styles.confirmContainer}\>

**CTO Confirmation:** Between the 3-second hold and the erasure, the user sees a **low-stimulus confirmation screen**. No more accidental "total wipes" from a shaky thumb.

**Important \#3: Restlessness Field Wiring**

**Files:** hooks/useOrchidSway.ts, app/(tabs)/journal.tsx

We are mapping raw accelerometer data to a clinical "restlessness" score (0.0 \- 1.0).

TypeScript

// hooks/useOrchidSway.ts  
export const useOrchidSway \= () \=\> {  
  const \[restlessness, setRestlessness\] \= useState(0);

  useEffect(() \=\> {  
    const subscription \= Accelerometer.addListener(({ x, y, z }) \=\> {  
      // Calculate magnitude delta (RMS approximation)  
      const magnitude \= Math.sqrt(x\*\*2 \+ y\*\*2 \+ z\*\*2);  
      const delta \= Math.abs(magnitude \- 1); // Gravity is 1g  
        
      // Normalize: 0 to 2.0 m/s² maps to 0.0 to 1.0  
      const score \= Math.min(delta / 2.0, 1.0);  
      setRestlessness(score);  
    });  
    return () \=\> subscription.remove();  
  }, \[\]);

  return { restlessnessScore: restlessness };  
};

// app/(tabs)/journal.tsx  
const { restlessnessScore } \= useOrchidSway();

const handleSave \= async () \=\> {

    score: currentScore,

  });  
};
