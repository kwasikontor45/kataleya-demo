\#\# Privacy Audit: The Honest Scope

Before the code, the audit question answered directly.

\*\*Q: Does anything in the current architecture violate the privacy promise?\*\*

\*\*A: No. But the promise requires precise language to be audit-proof.\*\*

\*\*Data that NEVER reaches any server:\*\*

| Data | Storage | Server-accessible |  
|------|---------|-------------------|  
| Mood logs | SQLite (device) | Never |  
| Journal entries | SQLite (device) | Never |  
| Circadian history | SQLite (device) | Never |  
| Sobriety start date | AsyncStorage (device) | Never |  
| Backup passphrase | Not stored | Never |

\*\*Data the sponsor relay holds in memory (intentional, opt-in):\*\*

| Field | Nature |  
|-------|--------|  
| Check-in date string | Presence signal ("Mon Mar 28 2026") |  
| Orchid stage label | Milestone phase ("seedling") |  
| Milestone count | Integer |  
| Presence signals | type: water or light only |

These exist because the sponsor channel is designed to share "I showed up today" with one trusted person. No mood score, no journal content, and no sobriety start date is ever sent to the server.

\*\*The corrected claim:\*\* \*"Your mood logs, journal entries, and all health-sensitive data physically cannot leave this device."\* This is provably true.

import { openDatabaseSync, type SQLiteDatabase } from 'expo-sqlite';  
import { Platform } from 'react-native';

let \_db: SQLiteDatabase | null \= null;

export function getDb(): SQLiteDatabase {  
  if (\_db) return \_db;  
  if (Platform.OS \=== 'web') {  
    throw new Error('SQLite unavailable on web');  
  }  
  \_db \= openDatabaseSync('kataleya.db', { enableChangeListener: false });  
  \_db.execSync(\`PRAGMA journal\_mode \= WAL;\`);  
  \_db.execSync(\`PRAGMA foreign\_keys \= ON;\`);  
  migrate(\_db);  
  return \_db;  
}

function migrate(db: SQLiteDatabase): void {  
  db.runSync(\`  
    CREATE TABLE IF NOT EXISTS mood\_logs (  
      id               TEXT PRIMARY KEY NOT NULL,  
      ts               INTEGER NOT NULL,  
      mood\_score       INTEGER NOT NULL CHECK (mood\_score BETWEEN 1 AND 10),  
      context          TEXT,  
      circadian\_phase  TEXT NOT NULL,  
      restlessness     REAL NOT NULL DEFAULT 0 CHECK (restlessness \>= 0 AND restlessness \<= 1),  
      correction       TEXT  
    )  
  \`);  
  db.runSync(\`CREATE INDEX IF NOT EXISTS idx\_mood\_ts ON mood\_logs (ts DESC)\`);

  db.runSync(\`  
    CREATE TABLE IF NOT EXISTS journal\_entries (  
      id               TEXT PRIMARY KEY NOT NULL,  
      ts               INTEGER NOT NULL,  
      body             TEXT NOT NULL,  
      mood\_score       INTEGER CHECK (mood\_score BETWEEN 1 AND 10),  
      circadian\_phase  TEXT NOT NULL  
    )  
  \`);  
  db.runSync(\`CREATE INDEX IF NOT EXISTS idx\_journal\_ts ON journal\_entries (ts DESC)\`);

  db.runSync(\`  
    CREATE TABLE IF NOT EXISTS circadian\_log (  
      id     INTEGER PRIMARY KEY AUTOINCREMENT,  
      ts     INTEGER NOT NULL,  
      phase  TEXT NOT NULL,  
      event  TEXT NOT NULL  
    )  
  \`);  
  db.runSync(\`CREATE INDEX IF NOT EXISTS idx\_circ\_ts ON circadian\_log (ts DESC)\`);  
}  
\`\`\`

\*\*What to verify:\*\*  
\- \`journal\_mode \= WAL\` — non-blocking reads during writes  
\- \`CHECK (mood\_score BETWEEN 1 AND 10)\` — rejected at database level, not application level  
\- No \`user\_id\` column exists in any table — the database has no concept of identity

export function useCircadian(): CircadianState {  
  const \[state, setState\] \= useState\<{ phase: CircadianPhase; blend: number }\>(computeState);  
  const timerRef \= useRef\<ReturnType\<typeof setTimeout\> | null\>(null);

  const schedule \= useCallback(() \=\> {  
    const next \= computeState();  
    setState(next);

    const minutes \= getCurrentMinutes();  
    const inTransition \= isInTransition(minutes);  
    const delayMs \= inTransition ? 10\_000 : msUntilNextMinute();

    timerRef.current \= setTimeout(schedule, delayMs);  
  }, \[\]);

  useEffect(() \=\> {  
    schedule();  
    return () \=\> { if (timerRef.current) clearTimeout(timerRef.current); };  
  }, \[schedule\]);

  const theme \= interpolateTheme(MorningBloom, MidnightGarden, state.blend);  
  return { phase: state.phase, blend: state.blend, theme,  
           phaseConfig: CIRCADIAN\_PHASES\[state.phase\], isGoldenHour: state.phase \=== 'goldenHour' };  
}  
\`\`\`

\*\*What to verify:\*\* No \`fetch()\` call. Phase computed from \`getCurrentMinutes()\` — a local clock read. The adaptive polling (10s during transition, 1-minute boundary otherwise) reduces JS thread wakeups during stable periods.

function scoreConfidence(n: number, mostRecentTs: number): number {  
  if (n \< 3\) return 0;  
  const sizeScore \= 1 \- 1 / (1 \+ n / 12);  
  const ageDays \= (Date.now() \- mostRecentTs) / 86\_400\_000;  
  const recency \= Math.exp(-ageDays / 45);  
  return Math.min(0.97, sizeScore \* recency);  
}

\*\*The encryption routine:\*\*  
\`\`\`typescript  
export async function encryptBackup(passphrase: string): Promise\<string\> {  
  // Collect all data from vaults  
  const \[moodLogs, journalEntries, circadianLog, sobrietyStart, name, substance\] \=  
    await Promise.all(\[  
      Sanctuary.getMoodLogs(5000),  
      Sanctuary.getJournalEntries(2000),  
      Sanctuary.getCircadianLog(365),  
      Surface.getSobrietyStart(),  
      Surface.getName(),  
      Surface.getSubstance(),  
    \]);

  const payload: BackupPayload \= {  
    v: BACKUP\_VERSION,  
    exportedAt: Date.now(),  
    sanctuary: { moodLogs, journalEntries, circadianLog },  
    surface: { sobrietyStart, name, substance },  
  };

  const plaintext \= JSON.stringify(payload);

  // Derive key — 100,000 PBKDF2-SHA256 iterations, 32-byte random salt  
  const salt \= CryptoJS.lib.WordArray.random(32);  
  const key  \= CryptoJS.PBKDF2(passphrase, salt, {  
    keySize: 256 / 32,  
    iterations: 100\_000,  
    hasher: CryptoJS.algo.SHA256,  
  });  
  const iv \= CryptoJS.lib.WordArray.random(16);

  // Encrypt (AES-256-CBC)  
  const encrypted \= CryptoJS.AES.encrypt(plaintext, key, {  
    iv, mode: CryptoJS.mode.CBC, padding: CryptoJS.pad.Pkcs7,  
  });

  // Authenticate — HMAC-SHA256 over ciphertext (Encrypt-then-MAC)  
  const mac \= CryptoJS.HmacSHA256(encrypted.ciphertext, key);

  const blob \= {  
    v: BACKUP\_VERSION,  
    alg: 'AES-256-CBC-PBKDF2-SHA256',  
    iters: 100\_000,  
    salt: salt.toString(CryptoJS.enc.Hex),  
    iv:   iv.toString(CryptoJS.enc.Hex),  
    mac:  mac.toString(CryptoJS.enc.Hex),  
    data: encrypted.ciphertext.toString(CryptoJS.enc.Base64),  
  };

  return JSON.stringify(blob);  
}  
\`\`\`

\*\*The decryption routine — HMAC check before decrypt:\*\*  
\`\`\`typescript  
export async function decryptBackup(blobStr: string, passphrase: string): Promise\<BackupPayload\> {  
  const blob \= JSON.parse(blobStr) as EncryptedBlob;

  const salt \= CryptoJS.enc.Hex.parse(blob.salt);  
  const iv   \= CryptoJS.enc.Hex.parse(blob.iv);  
  const key  \= deriveKey(passphrase, salt);  // same 100k PBKDF2 derivation

  // Verify MAC — wrong passphrase caught here, before any decryption  
  const ciphertext \= CryptoJS.enc.Base64.parse(blob.data);  
  const expectedMac \= CryptoJS.HmacSHA256(ciphertext, key).toString(CryptoJS.enc.Hex);  
  if (expectedMac \!== blob.mac) {  
    throw new Error('Incorrect passphrase or corrupt backup');  
  }

  // Decrypt only if MAC passes  
  const decrypted \= CryptoJS.AES.decrypt(  
    { ciphertext } as CryptoJS.lib.CipherParams,  
    key,  
    { iv, mode: CryptoJS.mode.CBC, padding: CryptoJS.pad.Pkcs7 }  
  );

  return JSON.parse(decrypted.toString(CryptoJS.enc.Utf8)) as BackupPayload;  
}

export async function restorePayload(payload: BackupPayload): Promise\<void\> {  
  await Sanctuary.burnAll();  // wipe existing data first

  if (payload.surface.sobrietyStart) await Surface.setSobrietyStart(payload.surface.sobrietyStart);  
  if (payload.surface.name)           await Surface.setName(payload.surface.name);  
  if (payload.surface.substance)      await Surface.setSubstance(payload.surface.substance);  
  await Surface.setOnboarded();

  // Restore oldest-first to preserve chronological ordering  
  for (const log of \[...payload.sanctuary.moodLogs\].reverse()) {  
    await Sanctuary.saveMoodLog({ ts: log.ts, moodScore: log.moodScore,  
      context: log.context, circadianPhase: log.circadianPhase,  
      restlessness: log.restlessness, correction: log.correction });  
  }  
  for (const entry of \[...payload.sanctuary.journalEntries\].reverse()) {  
    await Sanctuary.saveJournalEntry({ ts: entry.ts, body: entry.body,  
      moodScore: entry.moodScore, circadianPhase: entry.circadianPhase });  
  }  
}

\*\*What to verify:\*\*  
\- \`encryptBackup\` makes no \`fetch()\` call. The entire function is local.  
\- \`decryptBackup\` makes no \`fetch()\` call. The entire function is local.  
\- HMAC check (\`expectedMac \!== blob.mac\`) happens before \`CryptoJS.AES.decrypt\` is called. Wrong passphrase produces wrong key → wrong HMAC → rejected before any decryption.  
\- \`Sanctuary.burnAll()\` is called before any restore data is written — clean slate.  
\- The passphrase is never stored, logged, or sent anywhere. It is used to derive a key, then exits scope.
