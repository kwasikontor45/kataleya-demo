// api-server/index.ts

setInterval(() \=\> {  
  const now \= Date.now();  
  let pruned \= 0;

  \[rateLimits, globalIPLimits\].forEach(registry \=\> {  
    Object.keys(registry).forEach(key \=\> {  
      if (now \> registry\[key\].reset) {  
        delete registry\[key\];  
        pruned++;  
      }  
    });  
  });

  if (pruned \> 0) {  
    console.log(\`\[GC\] Pruned ${pruned} expired rate limit entries at ${new Date().toISOString()}\`);  
  }
