// hooks/useSponsorConnection.ts
import { useState, useEffect } from 'react';
import * as Crypto from 'expo-crypto';
import { Fortress } from '@/utils/storage';

interface SponsorKeys {
  userKeyPair: Crypto.CryptoKeyPair;
  sponsorPublicKey: Uint8Array;
  sharedSecret?: Crypto.CryptoKey;
}

export function useSponsorConnection() {
  const [connectionState, setConnectionState] = useState<'disconnected' | 'inviting' | 'connected'>('disconnected');
  
  const generateInvite = async (): Promise<string> => {
    // Generate Ed25519 keypair for E2E encryption
    const keyPair = await Crypto.generateKeyPairAsync(
      Crypto.CryptoAlgorithm.ED25519
    );
    
    // Store private key in Fortress (SecureStore)
    await Fortress.set('USER_PRIVATE_KEY', 
      Array.from(keyPair.privateKey).join(',')
    );
    
    // Create invite payload: public key + user ID + timestamp
    const invitePayload = {
      publicKey: Array.from(keyPair.publicKey),
      userId: await generateUserId(),
      expires: Date.now() + (7 * 24 * 60 * 60 * 1000), // 7 days
    };
    
    // Encode as visual glyph (QR-like pattern) + short code
    const visualGlyph = encodeToGlyph(invitePayload);
    const shortCode = generateShortCode(invitePayload);
    
    setConnectionState('inviting');
    return JSON.stringify({ visualGlyph, shortCode });
  };
  
  const acceptInvite = async (sponsorPublicKey: Uint8Array): Promise<void> => {
    // Derive shared secret using X25519
    const userPrivateKey = await Fortress.get('USER_PRIVATE_KEY');
    const sharedSecret = await deriveSharedSecret(
      userPrivateKey,
      sponsorPublicKey
    );
    
    // Store connection state
    await Fortress.set('SPONSOR_PUBLIC_KEY', 
      Array.from(sponsorPublicKey).join(',')
    );
    await Fortress.set('SHARED_SECRET',
      Array.from(sharedSecret).join(',')
    );
    
    setConnectionState('connected');
  };
  
  return {
    generateInvite,
    acceptInvite,
    connectionState,
  };
}
