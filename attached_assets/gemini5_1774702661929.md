export const useInsights \= () \=\> {  
  const \[insights, setInsights\] \= useState\<Insight\[\]\>(\[\]);  
  const \[metadata, setMetadata\] \= useState({   
    dataAgeDays: 0,   
    hasEnoughData: false   
  });

  useEffect(() \=\> {  
    const compute \= async () \=\> {  
      const logs \= await Sanctuary.getRecentMoodLogs(90); // Last 90 days  
        
      if (logs.length \=== 0) {  
        setMetadata({ dataAgeDays: 0, hasEnoughData: false });  
        return;  
      }

      const latestLogTs \= logs\[0\].ts;  
      const ageInDays \= Math.floor((Date.now() \- latestLogTs) / (1000 \* 60 \* 60 \* 24));  
        
      setMetadata({  
        dataAgeDays: ageInDays,  
        hasEnoughData: logs.length \>= 7 // Clinical threshold for pattern detection  
      });

      // ... existing confidence-gated pattern logic ...  
    };  
    compute();  
  }, \[\]);

  return { insights, ...metadata };  
};

#### **2\. Updated UI: app/(tabs)/insights.tsx**

The UI now handles the "cold start" and "stale data" states gracefully using the existing theme tokens.

TypeScript

// app/(tabs)/insights.tsx  
import { useInsights } from '../../hooks/useInsights';  
import { Tokens } from '../../constants/Theme';

export default function InsightsScreen() {  
  const { insights, hasEnoughData, dataAgeDays } \= useInsights();

  if (\!hasEnoughData) {  
    return (  
      \<View style={styles.emptyContainer}\>  
        \<Text style={{ color: Tokens.TextSecondary, textAlign: 'center' }}\>  
          Log your mood for a few days and patterns will start to appear here.  
        \</Text\>  
      \</View\>  
    );  
  }

  return (  
    \<ScrollView style={styles.container}\>  
      {insights.map(insight \=\> \<InsightCard key={insight.id} data={insight} /\>)}  
        
      {dataAgeDays \> 30 && (  
        \<View style={styles.staleNotice}\>  
          \<Text style={{ color: Tokens.TextTertiary, fontSize: 12 }}\>  
            Based on your logs from {dataAgeDays} days ago. Log today to refresh.  
          \</Text\>  
        \</View\>  
      )}  
    \</ScrollView\>  
  );  
}

### **CTO Final Audit & Response to Constraints:**

**Q: Do any notification payloads contain mood data or journal content?**

**A: Absolutely not.** All notification strings are hardcoded constants within the app binary ("Time for your evening check-in", "New Milestone reached"). Because we use expo-notifications locally, no user data—not even the fact that a user *is* in recovery—is sent to Apple (APNs) or Google (FCM) servers. The notification is triggered by a local timer looking at a local SQLite database.
