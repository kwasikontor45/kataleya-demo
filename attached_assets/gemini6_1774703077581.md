export const useSponsorChannel \= () \=\> {  
  const \[isSyncing, setIsSyncing\] \= useState(false);  
  const \[hasPending, setHasPending\] \= useState(false); // Fix: State-driven, not sync-call

  const checkQueueStatus \= async () \=\> {  
    const raw \= await Surface.get(PENDING\_KEY);  
    setHasPending(\!\!raw);  
  };

  const replayPendingCheckIn \= async () \=\> {  
    const raw \= await Surface.get(PENDING\_KEY);  
    if (\!raw) {  
      setHasPending(false);  
      return;  
    }

    const pending \= JSON.parse(raw);  
    const isExpired \= Date.now() \- pending.ts \> TTL\_MS;

    if (isExpired) {  
      await Surface.delete(PENDING\_KEY);  
      setHasPending(false);  
      return;  
    }

    try {  
      setIsSyncing(true);  
      await sendCheckInToServer(pending.data);  
      await Surface.delete(PENDING\_KEY);  
      setHasPending(false);  
    } catch (e) {  
      setHasPending(true);  
    } finally {  
      setIsSyncing(false);  
    }  
  };

  useEffect(() \=\> {  
    checkQueueStatus(); // Initial check on mount  
    if (isConnected) replayPendingCheckIn();  
  }, \[isConnected\]);

  return { performCheckIn, isSyncing, hasPending };  
};
